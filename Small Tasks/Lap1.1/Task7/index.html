#include<stdio.h>
#include<conio.h>

int exp(int base, int power) {
    if (power == 0) {
        return 1;
    } else {
        power--;
        return base * exp(base, power);
    }
}

main() {
    int x = 2, y = 3;
    clrscr();
    printf("\n %d ^ %d = %d", x, y, exp(x, y));
    getch();
    return 0;
}
